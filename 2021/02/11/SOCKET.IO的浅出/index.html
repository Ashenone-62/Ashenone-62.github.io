引言

SOCKET.IO是一个库，其又分为client客户端库与server服务端库。前者在前端配置使用，后者在后端配置使用。SOCKET.IO可以让我们的浏览器建立实时，双向和基于事件的通信，由此我们可以基于SOCKET.IO开发聊天APP、点餐APP、快递APP等等一些需要客户端和服务端建立通信连接的APP。

WebSocket与SOCKET.IO

WebScoket：是一种让客户端和服务器之间能进行双向实时通信的技术，是H5的一个协议规范，本质上是个基于TCP的协议，它通过HTTP/HTTPS协议发送一条特殊的请求进行握手后创建了一个TCP连接，此后浏览器/客户端和服务器之间便可以通过此连接来进行双向实时通信。使用需要浏览器支持，虽然到2020年的统计已有97%的浏览器支持，但是对于一些超古代和冷门浏览器依旧会面临不兼容的问题。

SOCKET.IO：SOCKET.IO是将WebSocket，Ajax和其它的通信方式全部封装成了统一的通信接口的库。比起WebScoket，使用起来不用担心兼容性问题，底层会自动选择最佳的通信方式。

使用

这里用Express和Vue两种框架演示。本地项目启动后，前端默认占用8080端口，后端默认占用3000端口。

构建项目

在目标文件下构建前后端项目。

//创建前端项目
vue create testApp_frontend
//创建后端项目
express testApp_server

安装依赖

分别cd进前后端的目录，前端安装socket.io-client，后端安装socket.io，推荐使用cnpm来安装，速度快。后端项目记得要执行cnpm/npm install一下，Express构建器默认会把自身所需依赖写入package.json，不会直接安装。

//前端项目目录内
cnpm/npm install socket.io-client
//后端项目目录内
cnpm/npm install socket.io

封装

前端

  • 在vue的静态文件夹assets下创建一个javascripts的文件夹，再在里面创建socket.io-client.js文件
  • 引入socket.io-client模块
  • 与服务器建立连接
  • 导出socket，之后在其他地方通过import...from...的方式就可以使用了
import io from 'socket.io-client'

let socket = io.connect('http://localhost:3000/');  //此处会返回一个对象给socket

export default socket

后端

  • 在Express的public的javascripts下创建socket.io.js文件
  • 创建一个空对象socketIO，方便之后导出
  • 创建getSocket函数，参数为服务器实例server
    • 引入socket.io模块并调用，参数为上面传入的server和{cors:true}（如果两个URL的protocol，port，host都相同的话，则这两个URL是同源。同源策略是重要的安全策略，浏览器默认会禁止所有跨域请求。此处我们前端的地址为http://localhost:8080/,后端是http://localhost:3000/，两者不是同源因此需要添加{cors:true}来解决跨域问题）
    • socketIO.io赋值给io
    • 监听connection事件
  • 把getSocket函数体给到socketIO.getSocket
  • 导出socketIO对象
let socketIO = {};

function getSocket(server){

    socketIO.io = require('socket.io')(server,{cors:true});//会返回一个对象给到socketIO.io

    let io = socketIO.io;

    io.on('connection',function(socket){
        console.log(socket)//成功触发默认会有一个socket对象传入
    })
        
}

socketIO.getSocket = getSocket;

module.exports = socketIO;
  • 在bin目录下的www中引入socketIO（具体位置见代码块）
  • 调用getSocket，把实例化好的server传入
var server = http.createServer(app);

//------上面是Express自己构建的------
//------下面是我们要操作的------

var socketIO = require("../socket.io.js");
socketIO.getSocket(server)
  • 如果正常，服务端的终端里会打印出socket对象

事件

监听与发送

socket.on('事件名',callback/Object) //监听
socket.emit('事件名',callback/Object) //发送
前端
<!-- App.vue -->
<!-- 前端浏览器的控制台会在点击按钮后打印WorldHello -->
<template>
  <div id="app">
    <button @click="emitEvent">
        点我发送事件
    </button>
  </div>
</template>

<script>
//导入socket
import socket from './assets/javascripts/socket.io'

export default {
  name: 'App',
  mounted(){
      socket.on('replyMsg',function(data){
        console.log(data)
    })
  },
  methods:{
      emitEvent:function(){
          socket.emit('sendMsg',{
              content: 'HelloWorld'
          })
      }
  }
}
</script>
后端
//socket.io.js
//后端终端会在点击按钮后打印HelloWorld
let socketIO = {};

function getSocket(server){

    socketIO.io = require('socket.io')(server,{cors:true});//会返回一个对象给到socketIO.io

    let io = socketIO.io;

    io.on('connection',function(socket){
         socket.on('sendMsg',function(data){
             console.log(data)
             socket.emit('replyMsg',{
                 content: 'WorldHello'
              })
         })
    })    
}

socketIO.getSocket = getSocket;

module.exports = socketIO;

广播

io.emit('事件名',callback/Object) //对所有人广播
前端
<!-- App.vue -->
<!-- 前端浏览器的控制台会在点击按钮后打印'全员广播 ' -->
<!-- 可以用多个浏览器来模拟多个连接来测试广播效果 -->
<template>
  <div id="app">
    <button @click="emitEvent">
        点我发送事件
    </button>
  </div>
</template>

<script>
//导入socket
import socket from './assets/javascripts/socket.io'

export default {
  name: 'App',
  mounted(){
      socket.on('all',function(data){
        console.log(data)
    })
  },
  methods:{
      emitEvent:function(){
          socket.emit('sendMsg',{
              content: 'isAll'
          })
      }
  }
}
</script>
后端
//socket.io.js
let socketIO = {};

function getSocket(server){

    socketIO.io = require('socket.io')(server,{cors:true});//会返回一个对象给到socketIO.io

    let io = socketIO.io;

    io.on('connection',function(socket){
         socket.on('sendMsg',function(data){
            if(data.content == "isAll"){
                io.emit('all',{
                     content: '全员广播'
                  })
             }
         })
    })
}

socketIO.getSocket = getSocket;

module.exports = socketIO;

断开

🚨🚨🚨注意这里演示的前端和后端是分别演示的，不是前后端联调🚨🚨🚨

socket.disconnect() //disconnect和connect一样也是可以监听的，事件名为disconnection
前端
<!-- App.vue -->
<!-- 点击按钮后当前连接断开 -->
<template>
  <div id="app">
    <button @click="discon">
        点我发送事件
    </button>
  </div>
</template>

<script>
//导入socket
import socket from './assets/javascripts/socket.io'

export default {
  name: 'App',
  methods:{
      discon:function(){
          socket.disconnect()
      }
  }
}
</script>
后端
//socket.io.js
//此处建立在前端会发送一个sendMsg事件的基础上
//点击按钮后触发sendMsg事件，然后当前所有连接断开
//可以用多个浏览器来模拟多个连接来测试断开
let socketIO = {};

function getSocket(server){

    socketIO.io = require('socket.io')(server,{cors:true});//会返回一个对象给到socketIO.io

    let io = socketIO.io;

    io.on('connection',function(socket){
         socket.on('sendMsg',function(){
            socket.disconnect()
         })
    })
        
}

socketIO.getSocket = getSocket;

module.exports = socketIO;

to方法

🚨🚨🚨每个socket对象都会有唯一的id属性，而to()中填写的是对方的socket.id🚨🚨🚨

🚨🚨🚨操作和发送大体一致，想办法拿到对方socket.id传入to()就行了，发送多人就作链式操作to()🚨🚨🚨

//对socket.id为socketId的连接发送事件
socket.to(socketID).emit('事件名',callback/Object)

//对socket.id为socketID1，socketID2，socketID3的连接发送事件
socket.to(socketID1).to(socketID2).to(socketID3).emit('事件名',callback/Object)

join方法

🚨🚨🚨当前连接加入一个名字为diyID的空间，此空间与其他空间隔绝，可以向加入了此空间的所有人广播🚨🚨🚨

let diyID = "xxxxx"
socket.join(diyID,callback/Object)
