给定一个非负索引 _k_，其中 k ≤ 33，返回杨辉三角的第 k 行。

我的思路

先按层处理，然后操作层内。已知层数，推出每层数的个数为层数+1。通过层层计算我们只需要保存上一层(父亲层)就能计算出下一层(儿子层

我的代码

/**
 * @param {number} rowIndex
 * @return {number[]}
 */
var getRow = function (rowIndex) {
  //定义儿子层，父亲层
  let pastRowArr = [];
  let currentRowArr = [];
  //循环层数
  for (let i = 0; i <= rowIndex; i++) {
    //初始化儿子层
    currentRowArr = [];
    //循环层内操作
    for (let j = 0; j <= i; j++) {
      //如果是第一个或者最后一个我们就进行相应操作
      if (j == 0 || j == i) {
        //第一个最后一个都是1
        currentRowArr.push(1);
        //三元运算符判断一下，当操作的是最后一个时我们把儿子层给父亲层保存一下，不然就等于他自己
        pastRowArr = j == i ? currentRowArr : pastRowArr;
      } else {
        //当不第一个或者最后一个时，我们把父亲层从索引值为0开始的两个相邻的数相加，然后放进儿子层
        currentRowArr.push(pastRowArr[j - 1] + pastRowArr[j]);
      }
    }
  }
  //最终我们得到处理好的儿子层(当前层)
  return currentRowArr;
};

/**
 * @param {number} rowIndex
 * @return {number[]}
 */
var getRow = function (rowIndex) {
  let rows = [];
  for (let i = 0; i <= rowIndex; i++) {
    let row = new Array(i + 1).fill(1);
    for (let j = 1; j < i; j++) {
      row[j] = rows[i - 1][j - 1] + rows[i - 1][j];
    }
    rows.push(row);
  }
  return rows[rowIndex];
};

/**
 * @param {number} rowIndex
 * @return {number[]}
 */
var getRow = function (rowIndex) {
  const idx = rowIndex + 1;
  const ans = new Array(idx);
  for (let i = 0; i < idx; ++i) {
    ans[i] = 1;
    let j = i - 1;
    while (j > 0) {
      ans[j] += ans[--j];
    }
  }
  return ans;
};
