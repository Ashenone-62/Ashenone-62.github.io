逆波兰表达式：

逆波兰表达式是一种后缀表达式，所谓后缀就是指算符写在后面。

平常使用的算式则是一种中缀表达式，如 ( 1 + 2 ) * ( 3 + 4 ) 。
该算式的逆波兰表达式写法为 ( ( 1 2 + ) ( 3 4 + ) * ) 。

逆波兰表达式主要有以下两个优点：

去掉括号后表达式无歧义，上式即便写成 1 2 + 3 4 + * 也可以依据次序计算出正确结果。
适合用栈操作运算：遇到数字则入栈；遇到算符则取出栈顶两个数字进行计算，并将结果压入栈中。

var evalRPN = function (tokens) {
  //定义处理加减乘除运算方案集合的对象
  const computedMark = {
    //加
    "+": (num1, num2) => {
      return num1 + num2;
    },

    //减
    "-": (num1, num2) => {
      return num1 - num2;
    },

    //乘
    "*": (num1, num2) => {
      return num1 * num2;
    },

    //除
    "/": (num1, num2) => {
      //这里用parseInt处理是因为如果除不尽需舍，不做处理会出现小数
      return parseInt(num1 / num2);
    },
  };

  //定义number数组，作为存放number的栈
  let numArr = [];
  //定义计算数组，作为存放需要计算的number的栈
  let computedArr = [];
  //定义计数器，用于管理计算数组
  let counter = 0;
  //定义位置，用于处理站内number与运算的特殊情况
  let posistion = 0;
  //定义上一次和本次是否计算，用于处理连续运算的情况
  let isComputed_last = false;
  let isComputed_now = false;

  //循环传入的逆波兰表达式
  for (let item of tokens) {
    //位置+1
    posistion++;
    //将tokens元素转化为number
    let itemTemplete = parseInt(item);
    //运算符转化后会变成NaN，用isNaN()判断，为false就是往栈内压number，true就是计算
    if (isNaN(itemTemplete) == false) {
      //---------往栈内压number---------

      //本次计算为false
      isComputed_now = false;
      //对于下一次操作来说，它的上一次没有计算也为false
      isComputed_last = false;

      //如果我们往计算数组里压入number小于2
      if (counter < 2) {
        //就向目标和计算数组同时压入number
        numArr.push(itemTemplete);
        computedArr.push(itemTemplete);
        //counter+1
        counter++;
      } else {
        //如果我们往计算数组里压入number大于（不可能出现）等于2
        //把计数设置为2
        counter = 2;
        //将计算数组的栈顶number赋值给栈底number
        //同时将栈顶number弹出
        //再把新的number压入计算数组和目标数组
        computedArr[0] = computedArr[1];
        computedArr.pop();
        numArr.push(itemTemplete);
        computedArr.push(itemTemplete);
      }
    } else {
      //---------如果是计算操作---------

      //本次肯定是计算
      isComputed_now = true;
      //本次相对于下一次也是计算了
      isComputed_last = true;

      //判断当前次计算是不是连续计算且计算数组的长度要小于2
      if (isComputed_now && isComputed_last && computedArr.length < 2) {
        //如果是

        //意味着连续运算，且除开上次的计算结果，目标数组只有一个number，且还没被放入计算number，如[2,5,4,*,+,...]

        //将计算数组的栈底向上移
        computedArr[1] = computedArr[0];
        //同时将目标数组的栈底放入（因为此时目标数组必然只有两个number）
        computedArr[0] = numArr[numArr.length - 2];
      }

      console.log("计算前", numArr, computedArr, counter);

      //初始化运算变量num1, num2
      let num1,
        num2 = 0;
      //将目标数组弹出两个number
      for (let i = 0; i < 2; i++) {
        numArr.pop();
      }
      //循环计算数组
      for (let numIndex in computedArr) {
        //将栈顶的给num1
        //栈底的给num2
        if (numIndex == 0) {
          num1 = computedArr[numIndex];
        } else {
          num2 = computedArr[numIndex];
        }
      }
      //通过键值来调用运算方案
      let computeRes = computedMark[item](num1, num2);
      //运算完后，置空运算数组
      computedArr = [];
      // 将结果放入目标的栈顶
      numArr.push(computeRes);

      // 这时候再判断一下特殊情况

      //如果tokens.length - posistion不等于1说明后续还有运算或者把number压入栈的操作
      if (tokens.length - posistion != 1) {
        //那我们只需要把结果压入计算数组里，后面不管是计算还是压入操作，别的地方会处理
        computedArr.push(numArr[numArr.length - 1]);
      } else {
        //相反tokens.length - posistion等于1说明后续只剩一位操作，那必然是运算操作结尾，目标数组里必然是只剩两个number
        //此时前面的步骤已经处理不到仅剩的两个number，我们就手动把它压入计算数组中
        computedArr.push(numArr[numArr.length - 2]);
        computedArr.push(numArr[numArr.length - 1]);
      }
      //更新一下couter
      counter = computedArr.length;

      console.log("计算后", numArr, computedArr, counter);
    }
  }
  //返回目标结果
  return numArr[0];
};

/**
 * @param {string[]} tokens
 * @return {number}
 */
var evalRPN = function (tokens) {
  if (!tokens || !tokens.length) return 0;
  let sign = ["+", "-", "*", "/"];
  let stack = [];
  for (let i = 0; i < tokens.length; i++) {
    let cur = tokens[i];
    if (sign.indexOf(cur) != -1) {
      let a = stack.pop();
      stack[stack.length - 1] = handleComputed(stack[stack.length - 1], a, cur);
    } else {
      stack.push(cur);
    }
  }
  return stack[0];
};

function handleComputed(a, b, sign) {
  let res = 0;
  switch (sign) {
    case "+":
      res = Number(a) + Number(b);
      break;
    case "-":
      res = a - b;
      break;
    case "*":
      res = a * b;
      break;
    case "/":
      res = parseInt(a / b);
      break;
  }
  return res;
}

/**
 * @param {string[]} tokens
 * @return {number}
 */
var evalRPN = function (tokens) {
  let n = tokens.length;
  if (n == 0) return 0;

  let stack = [];

  for (let i = 0; i < n; i++) {
    if (
      tokens[i] != "+" &&
      tokens[i] != "-" &&
      tokens[i] != "*" &&
      tokens[i] != "/"
    ) {
      stack.push(Number(tokens[i]));
    } else {
      let num1 = stack.pop();
      let num2 = stack.pop();
      switch (tokens[i]) {
        case "+":
          stack.push(num1 + num2);
          break;
        case "-":
          stack.push(num2 - num1);
          break;
        case "*":
          stack.push(num1 * num2);
          break;
        case "/":
          stack.push(parseInt(num2 / num1));
          break;
      }
    }
  }
  return stack[0];
};
