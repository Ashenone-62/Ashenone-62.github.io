「LeetCode」132模式

Words count 4.4k Reading time 4 mins. Read count 0

  • LeetCode
  • Normal

给定一个整数序列：a1, a2, …, an，一个 132 模式的子序列 ai, aj, ak 被定义为：当 i < j < k 时，ai < ak < aj。设计一个算法，当给定有 n 个数字的序列时，验证这个序列中是否含有 132 模式的子序列。

我的思路

思路是枚举三个数，妥妥的超时。定义一个函数每次放入三个数来比较，如果不符合 132 模式且数组还有剩余，就把栈顶弹出放入一个新的在比较，如此反复操作完还没符合 132 模式，就弹出第二位的数，然后反复操作，如果还没有符合的，就将上回未修改的的数组的第一个数删掉（也就是 12345——>123,124,125,134,134,145,所有情况枚举完了，下一轮就是 2345 开始了），再把上面操作如此重复循环，直到有结果为止。

我的代码

var find132pattern = function (nums) {
  //初始化一个数组保存上轮循环的数组，如12345，第一轮是1....开始，第二轮是2.....
  let lastArr = [];
  //初始化目标值
  let is132Model = false;
  //定义一个比较数组
  let compareArr = [];
  //循环计数器
  let counter = 0;
  //定义一个函数处理数组
  const doing = function (currentArr) {
    //初始化一下循环数组与计数器
    compareArr = [];
    counter = compareArr.length;
    //循环当前的数组
    currentArr.forEach((item, index) => {
      //当比较数组长度等于3的时候
      if (counter == 3) {
        //是

        //判断一下比较数组是否满足132模式
        if (compareArr[0] < compareArr[2] && compareArr[2] < compareArr[1]) {
          //满足就置目标值为true
          is132Model = true;
          return;
        } else {
          //不满足就把栈顶元素弹出
          //放入当前数组后面的数
          //更新计数器
          compareArr.pop();
          compareArr.push(item);
          counter = compareArr.length;
          //操作完后，再次判断一下符不符合132模式
          if (compareArr[0] < compareArr[2] && compareArr[2] < compareArr[1]) {
            is132Model = true;
            return;
          }
          //还不满足132模式
          //当此时是最后一个数，且当前的数组不是最后三个数组成的数组的时候（如原始数组是12345已经枚举到145已经是最后一次枚举了，就不满足此时的条件了）
          if (index + 1 == currentArr.length && currentArr.length > 3) {
            //将第一个数后移
            //然后删掉第一个数（如12345——>123,124,125后，肯定下次就是134了，相当于下次遍历的数组是1345）
            currentArr[1] = currentArr[0];
            currentArr.shift();
            //将操作完的数组传给doing
            doing(currentArr);

            //当此时是最后一个数，且当前的数组是最后三个数组成的数组的时候（如原始数组是12345已经枚举到145已经是最后一次枚举了，就满足此时的条件了，那么下一次肯定就是循环2345——>234,245）
          } else if (index + 1 == currentArr.length && currentArr.length <= 3) {
            lastArr.shift();
            //同样是深拷贝，不然修改会联动，导致结果错误
            doing(lastArr.concat());
          }
        }
      } else {
        //否

        //如果比较数组还没有3个，就放入数
        compareArr.push(item);
        counter = compareArr.length;
        //当放入的数是最后一个的时候，上面的判断是否符合132模式的代码就不管用了，需要在这儿额外判断
        if (
          compareArr[0] < compareArr[2] &&
          compareArr[2] < compareArr[1] &&
          compareArr.length == 3
        ) {
          is132Model = true;
          return;
        }
        //此时必然是最后一个数，那最后一个数必然是最后一种结果（如12345的145），把保存上一次循环第一个数删掉（这里操上一次的数组可以省去操作当前数组完还要更新上一次数组的操作，因为已经是当轮的最后一次了嘛），开启新循环
        if (index + 1 == currentArr.length && currentArr.length == 3) {
          lastArr.shift();
          doing(lastArr.concat());
        } else {
          return;
        }
      }
    });
  };
  //去重，如果只有两个数必然是不符合132模式的
  if (new Set(nums).size > 2) {
    let aimArr = [];
    //如果符合再次去重，如122344445，相当于12345，这里ti'qian
    for (let index = 0; index < nums.length; index++) {
      if (nums[index] != nums[index + 1]) {
        aimArr.push(nums[index]);
      }
    }
    //将处理好的数组深拷贝给保存上一次的数组
    lastArr = aimArr.concat();
    //调用doing
    doing(aimArr);
  }
  //返回目标值
  return is132Model;
};

官答

单调栈——枚举 1 或者 2 或者 3（对应 i，j，k），此处是枚举 1

var find132pattern = function (nums) {
  const n = nums.length;
  const candidate_k = [nums[n - 1]];
  let max_k = -Number.MAX_SAFE_INTEGER;

  for (let i = n - 2; i >= 0; --i) {
    if (nums[i] < max_k) {
      return true;
    }
    while (
      candidate_k.length &&
      nums[i] > candidate_k[candidate_k.length - 1]
    ) {
      max_k = candidate_k[candidate_k.length - 1];
      candidate_k.pop();
    }
    if (nums[i] > max_k) {
      candidate_k.push(nums[i]);
    }
  }
  return false;
};
