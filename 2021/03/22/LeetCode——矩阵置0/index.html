LeetCode——矩阵置0

给定一个 m x n 的矩阵，如果一个元素为 0 ，则将其所在行和列的所有元素都设为 0。

我的思路

将矩阵每行遍历一遍，去找0的列位置，找到之后单独用一个数组管理起来，只要有0我们最后的时候就直接把整行置为0。然后去设置列，将每行中的列位置（刚刚记录的）数字置为0。

我的代码

var setZeroes = function(matrix) {
    //定义一个数组管理每列0的位置
    let posistion_column = [];
    //定义一个number保存每行的长度
    let matrix_child_lenth = matrix[0].length;
    
    //行循环，循环传入的矩阵
    for (let row in matrix) {
        //定义一个变量来表示是否此行存在0
        let isZero = false;
        //列循环，循环每行找到0的位置，把他放入上面提到的数组中，同时把isZero置为true
        for (let column in matrix[row]) {
            if (matrix[row][column] == 0) {
                isZero = true;
                posistion_column.push(column);
            }
        }
        //再列循环完后，判断一下有无0，有0就直接把整行换为长度相同的每个元素为0的数组
        if (isZero) {
            matrix[row] = new Array(matrix_child_lenth).fill(0);
        }
    }
    //修改列，再次循环行，同时循环位置数组，将对应位置改为0
    for (let row in matrix) {
        for (let column of posistion_column) {
            matrix[row][column]=0
        }
    }

    return matrix
};

LC时间第一

var setZeroes = function(matrix) {
    if (matrix.length === 0) return matrix
    let m = matrix.length, n = matrix[0].length, row = [], col = [];
    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (matrix[i][j] === 0) {
                row.findIndex(item => item === i) === -1 && row.push(i);
                col.findIndex(item => item === j) === -1 && col.push(j);
            }
        }
    }
    row.forEach(item => {
        matrix[item] = new Array(n).fill(0);
    })
    col.forEach(item => {
        for (let i = 0; i < m; i++) {
            matrix[i][item] = 0;
        }
    })
};

/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var setZeroes = function(matrix) {
    let rLen = matrix.length;
    let cLen = matrix[0].length;
    let col0Deal = false;
    // 标记
    for(let i = 0;i<rLen;i++){
        if(matrix[i][0] === 0) col0Deal = true;
        for(let j = 1;j<cLen;j++){
            if(matrix[i][j] === 0) {
                matrix[i][0] = 0;
                matrix[0][j] = 0;
            }
        }
    }
    // 处理
    for(let i = rLen - 1;i>=0;i--){
        for(let j = cLen - 1;j > 0;j--){
            if(matrix[i][0] === 0 || matrix[0][j] === 0) matrix[i][j] = 0;
        }
    }
    if(col0Deal) for(let i = 0;i<rLen;i++) matrix[i][0] = 0;
};
