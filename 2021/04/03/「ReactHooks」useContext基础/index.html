前言

useContext 让开发者能更方便的处理隔多代组件的传值问题。useContext 可以获取父级组件传递过来的 context 值，这个当前值就是最近的父级组件 Provider 设置的 value 值，useContext 参数必须是 Context 对象本身。

初识

contextType

/* 创建Context对象 */
const Context = createContext()
class ClassDemo extends Component {
  render() {
    /* this.context 可以访问到 ClassDemo 的contextType */
    let value = this.context;
    return (
        <div>
            contextType —— my name is { value.name }
        </div>
    )
  }
}
/* 将 ClassDemo 的 contextType 属性赋值为 Context 对象的值 */
ClassDemo.contextType = Context;

function App() {
    return (
        <div>
            <Context.Provider value={{ name:'alien' , age:18 }} >
                <ClassDemo />
            </Context.Provider>
        </div>
    )
}

Context.Consumer

/* 创建Context对象 */
const Context = createContext()
function DemoContextConsumer() {
    return (
        /* 通过Context.Consumer获取Provider的值 */
        <Context.Consumer>
            { (value) =>
                <div>
                    Context.Consumer —— my name is { value.name }
                </div>
            }
        </Context.Consumer>
    )
}
function App() {
    return (
        <div>
            <Context.Provider value={{ name:'alien' }} >
                <DemoContextConsumer />
            </Context.Provider>
        </div>
    )
}

useContext

/* 创建Context对象 */
const Context = createContext()
function DemoUseContext() {
     /* 通过useContext获取Provider的值 */
    const value = useContext(Context)

    return (
        <div>
            useContext —— my name is { value.name }
        </div>
    )
}
function App() {
    return (
        <div>
            <Context.Provider value={{ name:'alien' }} >
                <DemoUseContext />
            </Context.Provider>
        </div>
    )
}

useContext 基础

写法

useContext 的写法为 useContext(Context),useContext 参数必须是 Context 对象本身，最后会返回当前 Context 的 value 。

  • 当组件上层最近的 <MyContext.Provider> 更新时，该 Hook 会触发重渲染，并使用最新传递给 MyContext的 provider 的 context value 值。即使祖先使用 React.memo 或 shouldComponentUpdate，也会在组件本身使用 useContext 时重新渲染，官方推荐使用 memoization 进行优化。

使用

import { useContext , createContext} from 'react'
/* 创建Context对象 */
const Context = createContext()

function DemoUseContext() {
    /* 将Context对象传入useContext，最终会返回Context的当前value */
    const value = useContext(Context)
    return (
        <div>
            useContext —— my name is { value.name }
        </div>
    )
}
function App() {
    return (
        <div>
            {/* 使用Context.Provider来传递value */}
            <Context.Provider value={{ name:'alien' }} >
                <DemoUseContext />
            </Context.Provider>
        </div>
    )
}
export default App
