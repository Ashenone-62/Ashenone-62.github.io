$的封装

//定义$符号
//传入选择器
let $ = function (selector) {
  //获取dom节点
  let doms = document.querySelectorAll(selector);
  //返回dom节点
  return doms;
};

添加 html 方法

let $ = function (selector) {
  let doms = document.querySelectorAll(selector);

  //通过__proto__添加原型方法html，传入一个value值
  doms.__proto__.html = function (value) {
    //循环获取的dom
    doms.forEach(function (element, index) {
      //把value给到dom的innerHTML方法
      element.innerHTML = value;
    });
  };

  return doms;
};

添加 css 方法

let $ = function (selector) {
  let doms = document.querySelectorAll(selector);

  //通过__proto__添加原型方法css，传入一个value值
  doms.__proto__.css = function (key, value) {
    //判断key是不是为字符串类型，因为此处只会传字符串或者对象
    //如$("div").css("backgroundColor","red")
    //如$("div").css({
    //      width:"300px",
    //      height:"300px"
    //     })

    if (typeof key == "string") {
      //如果为字符串就循环获取的dom节点
      doms.forEach(function (element, index) {
        //此处key就为样式的属性名，value为属性值
        element.style[key] = value;
      });
    } else {
      //否则就循环传入的对象
      for (let i in key) {
        //此处i为属性名，key[i]为属性值(参考——对象[键]来理解)
        doms.forEach(function (element, index) {
          //与上面一样
          element.style[i] = key[i];
        });
      }
    }
  };

  return doms;
};
