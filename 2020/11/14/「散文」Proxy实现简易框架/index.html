Proxy 基础

  • 概念

可以理解成，在目标对象之前架设一层“拦截”，外界对该对象的访问，都必须先通过这层拦截，因此提供了一种机制，可以对外界的访问进行过滤和改写。

  • 案例
let userObj = {
  name: "张三",
  age: 18,
};

//创建一个新的Proxy对象，第一个参数传要代理的对象，第二个参数传要执行的内容
let proxyUserObj = new Proxy(user, {
  get(target, attr) {
    console.log(target); //对象
    console.log(attr); //属性
    return target[attr];
  },

  set(target, attr, value) {
    console.log(target); //对象
    console.log(attr); //属性
    console.log(value); //值
    target[attr] = value;
  },
});

var age = proxyUserObj.age;
proxyUserObj.age = 20;

简易前端框架原理

通过 JavaScript 快速创建 Dom 对象

//创建DOM的代理对象，此处不需要给任何对象代理所以传一个空对象进去，第二个参数照旧
let DOM = new Proxy(
  {},
  {
    get(target, attr) {
      //通过createElement创建对应attr的DOM元素
      let domObj = document.createElement(attr);

      //这里最终返回的是domObj，function里是对domObj的操作
      //第一个参数attrs传入的是一个数组，里面写的是DOM元素的属性
      //第二个参数children为追加的内容，此处只做了string的操作，由于可能追加多个内容，为了全部获取我们需要拓展运算符...
      return function (attrs, ...children) {
        //通过for in循环设置所有的属性
        for (let key in attrs) {
          domObj.setAttribute(key, attrs[key]);
        }
        //通过for循环创建文本节点
        for (let i = 0; i < children.length; i++) {
          if (typeof children[i] == "string") {
            children[i] = document.createTextNode(children[i]);
          }
          //在domObj中追加子元素
          domObj.appendChild(children[i]);
        }
        //将操作好的domObj返回
        return domObj;
      };
    },
  }
);
